# 순열 (Permutation)
- 순서가 정해진 임의의 집합을 다른 순서로 섞는 연산 (경우의 수는 n!)
  - ex) 1,2,3이 있을 때 다른 순서로 섞는 연산은 3!로 총 6가지
  - ~순서라는 말이 들어가면 순열

## 재귀함수를 이용한 순열

```cpp
#include <algorithm>
#include <iostream>
using namespace std;
int a[3] = {1, 2, 3};
int n = 3, r = 3;
void print(){
    for (int i = 0; i < r; i++){
        cout << a[i] << " ";
    }
    cout << "\n";
}

void makePermutation(int n, int r, int depth){
    if (r == depth){
        print();
        return;
    }
    for (int i = depth; i < n; i++){
        swap(a[i], a[depth]);
        makePermutation(n, r, depth+1);
        swap(a[i], a[depth]);
    }
    return;
}
int main(){
    makePermutation(n, r, 0);
    return 0;
}
```
